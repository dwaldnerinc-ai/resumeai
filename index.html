Safe Escape Function --- function escapeHTML(s) { return String(s) .replace(/&/g, "&") .replace(//g, ">") .replace(/\"/g, """) .replace(/'/g, "'"); } // --- 2. ${escapeHTML(data.name)}

` : ""}

${escapeHTML(section.title)}
`; let inList = false; section.items.forEach(item => { if (item.type === "job") { if (inList) { html += ""; inList = false; } html += `
${escapeHTML(item.text)}
`; } else if (item.type === "bullet") { if (!inList) { html += "
    "; inList = true; } html += `
  • ${escapeHTML(item.text)}
    • `; } else { if (inList) { html += "
"; inList = false; } html += `
${escapeHTML(item.text)}
`; } }); if (inList) html += ""; html += `
`; }); const footerText = ci.footer || "Professional Portfolio"; html += `
${getFormattedDate()} | ${escapeHTML(footerText)}
`; return `
${html}
`; } // --- 5. Main Builder --- function buildPDFHTML(text, ci = {}, theme = "modern") { const parsed = parseResume(text); return { html: renderHTML(parsed, ci, theme), data: parsed }; } // --- 6.
